Rmd chunk 中循环输出需要使用cat函数

knitr::opts_chunk$set(echo = TRUE)
library(dplyr)
library(lubridate)
library(knitr)
library(kableExtra)
library(htmltools)
for (i in 1:3) {
  cat("\n")
  cat("### ", sprintf('第%d次', i), "\n")
  
  tab11 = iris %>%
    head(10) %>%
    kable(row.names = F) %>%
    kable_styling() %>%
    row_spec(0, background = 'grey') %>%
    row_spec(10, background = '#E6C1C1')
  
  cat(tab11)
  cat('\n')
  cat('\n')
  cat('\n')
}
